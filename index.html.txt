<script>
  const BENCH = {
    practiceAreas:{
      "Family Law, Divorce":{cpcLow:15,cpcHigh:60,lp:0.07},
      "Personal Injury":{cpcLow:60,cpcHigh:180,lp:0.05}
    },
    market:{major:1.35,mid:1.1,small:.9}
  };

  const $ = id => document.getElementById(id);

  function updateBudgetLabel(){
    $("budgetLabel").textContent = `USD ${Number($("budget").value).toLocaleString()}`;
  }

  function run(){
    const area = BENCH.practiceAreas[$("practiceArea").value];
    const m = BENCH.market[$("marketSize").value];
    const budget = Number($("budget").value);
    const close = $("closeRate").value/100;
    const acv = Number($("acv").value);
    const cpc = ((area.cpcLow+area.cpcHigh)/2)*m;

    const leads = (budget/cpc)*area.lp;
    const clients = leads*close;
    const revenue = clients*acv;
    const roi = ((revenue-budget)/budget)*100;

    $("kpiLeads").textContent = Math.round(leads);
    $("kpiClients").textContent = clients.toFixed(1);
    $("kpiRevenue").textContent = `USD ${Math.round(revenue).toLocaleString()}`;
    $("kpiRoi").textContent = `${Math.round(roi)}%`;

    const warn = $("assumptionWarnings");
    warn.innerHTML="";
    if(close>0.6){
      warn.innerHTML+=`<div class="badge warn">Close rate above 60 percent is optimistic.</div>`;
    }
    if(area.lp>0.25){
      warn.innerHTML+=`<div class="badge warn">Landing page conversion above 25 percent is uncommon.</div>`;
    }
  }

  function init(){
    Object.keys(BENCH.practiceAreas).forEach(p=>{
      const o=document.createElement("option");
      o.value=p; o.textContent=p;
      $("practiceArea").appendChild(o);
    });
    updateBudgetLabel();
    ["practiceArea","marketSize","budget","closeRate","acv"].forEach(id=>{
      $(id).addEventListener("input",()=>{updateBudgetLabel();run();});
    });
    run();
  }

  init();
</script>
